% \newcommand-定义命令
% 命令只能由字母组成，不能以\end 开头
% \newcommand{<命令>}[<参数个数>][<首参数默认值>]{<具体定义>}

% \newcommand 可以是简单的字符替换，例如
\newcommand{\PRC}{People's Republic of \emph{China}}

% \newcommand 也可以使用参数
% 参数个数可以从1到9，使用时用#1，#2，......，#9 表示
\newcommand{\loves}[2]{#1 喜欢 #2}
\newcommand{\hateby}[2]{#2 不受 #1 喜欢}

% \newcommand 的参数也可以有默认值
% 只能指定第一个参数的默认值
% 第一个参数成为可选参数
\newcommand{\love}[3][喜欢]{#2#1#3}

% \renewcommand-重定义命令
% 与newcommand的用法完成一致，但只能用于已有命令
% \renewcommand{<命令>}[<参数个数>][<首参数默认值>]{<具体定义>}
\renewcommand{\abstractname}{我的内容简介}

% 定义和重定义环境
% \newenvironment{name}[args][default]{begdef}{enddef}
% \renewenvironment{name}[args][default]{begdef}{enddef}

\newenvironment{myabstract}[1][摘要]

% 环境参数只有<环境前定义>中可以使用参数，
% 环境后定义不能使用参数
% 如果需要，可以把前面得到的参数保存在一个命令中，在后面使用

\newenvironment{myqotation}[1]%
{\newcommand\quotesource{#1}%
	\begin{quotation}}%
		{\par\hfill---《\textit{\quotesource}》\end{quotation}}

    \end{myabstract}
