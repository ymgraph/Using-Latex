     \section{多行公式}
     \begin{gather}
     	a + b = b + a \\
     	a \times b = b \times a 
     \end{gather}
     % 取消编号
     \begin{gather*}
        a + b = b + a \\
        a \times b = b \times a 
     \end{gather*}
     % 在\\前使用\notag 阻止编号
     \begin{gather}
     	a + b = b + a \notag \\
     	a \times b = b \times a 
     \end{gather}
     
     % align 和 align* 环境(用& 进行对齐)  
     %带编号
     \begin{align}
     	x & = t + \cos t + 1 \\
     	y & = 2\sin t
     \end{align} 
     %不带编号  
     \begin{align*}
         x & = t + \cos t + 1 \\
         y & = 2\sin t
     \end{align*}
     
     % split环境将一个公式多行排版
     \begin{equation}
     \begin{split}
     \cos 2x &= \cos^2 x - \sin^x \\
     &= 2\cos^2 x - 1
     \end{split}
     \end{equation}
